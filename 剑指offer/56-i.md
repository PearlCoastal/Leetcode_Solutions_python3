class Solution:
    def singleNumbers(self, nums: List[int]) -> List[int]:

        dic = collections.defaultdict(int)

        for num in nums:
            dic[num] += 1
        ans = []
        for key, value in dic.items():
            if dic[key] == 1:
                ans.append(key)
        return ans
