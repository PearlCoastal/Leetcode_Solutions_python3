```python
import heapq
class Solution:
    def getLeastNumbers(self, arr: List[int], k: int) -> List[int]:
        if not arr or not k:
            return []
        heap = []
        for i in range(k):
            heapq.heappush(heap, -arr[i])
        
        for i in range(k, len(arr)):
            if arr[i] < -heap[0]:
                heapq.heappop(heap)
                heapq.heappush(heap, -arr[i])
        
        ans = []
        while heap:
            ans.append(-heapq.heappop(heap))
        return ans
```
