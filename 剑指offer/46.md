```python
class Solution:
    def translateNum(self, num: int) -> int:

        if num < 10:
            return 1
        s = str(num)
        dp = [0 for i in range(len(s) + 1)]
        dp[0], dp[1] = 1, 1

        for i in range(2, len(s) + 1):
            if "10" <= s[i - 2: i] <= "25":
                dp[i] = dp[i - 1] + dp[i - 2]
            else:
                dp[i] = dp[i - 1]
        return dp[-1]
```
