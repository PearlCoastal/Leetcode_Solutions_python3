```python
class Solution:
    def search(self, nums: List[int], target: int) -> int:
        if not nums:
            return 0
        
        def bi_left(nums: List[int], target: int) -> int:
            left, right = 0, len(nums) - 1
            while left <= right:
                mid = (left + right) // 2
                if nums[mid] == target:
                    right = mid - 1
                elif nums[mid] < target:
                    left = mid + 1    
                else:
                    right = mid - 1
            return left
            
        def bi_right(nums: List[int], target: int) -> int:
            left, right = 0, len(nums) - 1
            while left <= right:
                mid = (left + right) // 2
                if nums[mid] == target:
                    left = mid + 1
                elif nums[mid] < target:
                    left = mid + 1     
                else:
                    right = mid - 1
            return right
            
        return bi_right(nums, target) - bi_left(nums, target) + 1
```
