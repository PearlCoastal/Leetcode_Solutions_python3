```python
class Solution:
    def kthLargest(self, root: TreeNode, k: int) -> int:
        
        self.ans = 0
        self.k = k

        def dfs(root):
            if not root or self.k == 0:
                return
            
            dfs(root.right) 
            self.k -= 1
            if self.k == 0:
                self.ans = root.val
            dfs(root.left)
        dfs(root)
        return self.ans
```
