class Solution:
    def numRescueBoats(self, people: [int], limit: int) -> int:

        people.sort()
        ans = 0

        while len(people) > 0:
            ans += 1
            if len(people) > 1 and people[-1] + people[0] <= limit:
                people.remove(people[0])
            people.remove(people[-1])

        return ans
# @lc code=end

class Solution:
    def numRescueBoats(self, people: [int], limit: int) -> int:

        people.sort()
        ans = 0
        left, right = 0, len(people) - 1
        while left <= right:
            if people[left] + people[right] <= limit:
                left += 1
            right -= 1
            ans += 1

        return ans




# @lc code=end
