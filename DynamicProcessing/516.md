```python
class Solution(object):
    def longestPalindromeSubseq(self, s):
        @cache
        def dp(left, right):
            if left == right:
                return 1
            elif left > right:
                return 0
            if s[left] == s[right]:
                return dp(left + 1, right - 1) + 2
            return max(dp(left + 1, right), dp(left, right - 1))

        return dp(0, len(s) - 1) 

```
