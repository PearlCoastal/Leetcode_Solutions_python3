# @lc code=start
class Solution:
    def findMinArrowShots(self, points: [[int]]) -> int:
        if not points:
            return 0
        points.sort()
        dp = [1] * len(points)
        for i in range(len(points)):
            for j in range(i - 1, -1, -1):
                if points[i][0] > points[j][1]:
                    dp[i] = max(dp[i], dp[j] + 1)
        return max(dp)
# @lc code=end
