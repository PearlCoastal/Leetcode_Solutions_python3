```python
class Solution:
    def smallestK(self, arr: List[int], k: int) -> List[int]:
        if k == 0:
            return list()
        h = [-x for x in arr[:k]]
        heapq.heapify(h)

        for i in range(k, len(arr)):
            if -h[0] > arr[i]:
                heapq.heappop(h)
                heapq.heappush(h, -arr[i])
        ans = [-x for x in h]
        return ans
```
