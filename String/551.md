```python
class Solution:
    def checkRecord(self, s: str) -> bool:

        A, L = 1, 3

        for i in range(len(s)):
            if s[i] == 'A':
                if A == 0:
                    return False
                A -= 1
            if s[i] == 'L':
                if i + 2 < len(s):
                    for j in range(i, i + 3):
                        if s[j] == 'L':
                            L -= 1
                if L <= 0:
                        return False
                else: 
                    L = 3
        return True
```
