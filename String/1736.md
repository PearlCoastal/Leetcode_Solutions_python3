class Solution:
    def maximumTime(self, time: str) -> str:
        time = list(time)
        for i in range(len(time)):
            if time[i] == '?':
                if i == 0:
                    if time[i + 1] == '?' or int(time[ i+1 ])<=3:
                        time[i] = '2'
                    else:
                        time[i] = '1'
                if i == 1:
                    if time[i - 1] == '0' or time[i - 1]=='1':
                        time[i] = '9'
                    else:
                        time[i] = '3'
                if i == 3:
                    time[i] = '5'
                if i == 4:
                    time[i] = '9'
        return ''.join(time)
